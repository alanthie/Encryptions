Example

Linux:
./crypto -h

./crypto encode  -i ./test.zip -o ./encoder_output/test.zip.encrypted -p ./puzzle.txt -q ./encoder_output/partial_puzzle.txt -u ./urls.txt -v 1 -s ./tmp/ -k 1 -f ./encoder_output/puzzle_full.txt
./crypto decode  -i ./encoder_output/test.zip.encrypted -p ./encoder_output/puzzle.txt.full -o ./decoder/output/test.zip.unencrypted -v c


Windows
crypto.exe encode  -i ./test.zip -o ./encoder_output/test.zip.encrypted -p ./puzzle.txt -q ./encoder_output/partial_puzzle.txt -u ./urls.txt -v 1 -s ./tmp/ -k 1 -f ./encoder_output/puzzle_full.txt
crypto.exe decode  -i ./encoder_output/test.zip.encrypted -p ./encoder_output/puzzle.txt.full -o ./decoder/output/test.zip.unencrypted -v c
